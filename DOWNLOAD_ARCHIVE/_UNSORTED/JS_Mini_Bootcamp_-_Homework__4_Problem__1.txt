//The First one works as intended.

function multiplyArguments() {
  var product = 1;
  
  if (arguments.length === 0) {
    return 0;
  }
  for (var i = 0; i < arguments.length; i++) {
    product *= arguments[i];
  }
  return product;
}

//The second one won't return 0 when no arguments have been passed

function multiplyArguments() {
  var product = 1;
  
  for (var i = 0; i < arguments.length; i++) {
    if (arguments.length === 0) {
      return 0;
    } else {
      product *= arguments[i];
    }
  }
  return product;
}
